#include<stdio.h>
#include<malloc.h>
#define TABLE_SIZE 10
#define NUMBER_SIZE 12

typedef struct Hash{
    char box[12];
}HashNode,*Hash;

int main()
{
    int i=0,j,k,num,tnum;
    Hash H,Temp;
    char input[121]={};
    H=(Hash)malloc(TABLE_SIZE*sizeof(HashNode));
    memset(H,0,TABLE_SIZE*sizeof(HashNode));//清空
    scanf("%s",input);
    do{
        num=input[i+10]-48;//尾号
        if((H[num]).box[0]=='\0') {
            memcpy((H[num]).box,&input[i],11);
        }
        else {
            k=1;//冲突次数
            while((H[(num+k*k)%TABLE_SIZE]).box[0]!='\0')//平方探测
                ++k;
            num=(num+k*k)%TABLE_SIZE;
            memcpy((H[num]).box,&input[i],11);
        }
        i+=NUMBER_SIZE;
    }while(input[i]!='\0');
     
    for(i=0;i<TABLE_SIZE;i++){
        if('\0'==(H[i]).box[0])
            printf("0,");
        else
            printf("%s,",(H[i]).box);
    }

}
